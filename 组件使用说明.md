- 日期选择
```js
data: [
  [
    {
      label: '有效期：',
      prop: 'startime_endtime',
      tag: {
        tagType: 'date-picker',
        attr: {
          type: 'datetimerange',
          placeholder: '请选择有效期',
          'range-separator': '至',
          'start-placeholder': '开始日期',
          'end-placeholder': '结束日期',
        },
      },
    }
  ]
],
handleSubmit: (data) => {
  const [startime, endtime] = this.$utils._DataTypeChange(data.startime_endtime);
  delete data.startime_endtime;
  return { ...data, startime, endtime };
},
```
